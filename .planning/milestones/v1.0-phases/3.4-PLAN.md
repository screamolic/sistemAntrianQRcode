### 1. Create Queue Entry Schema Extension

**File:** `src/lib/queue-entries.ts`

```typescript
import { db } from './prisma';
import { QueueEntryStatus } from '@prisma/client';

export async function getQueueEntries(queueId: string) {
  return db.queueEntry.findMany({
    where: { queueId },
    orderBy: [
      { position: 'asc' },
      { createdAt: 'asc' },
    ],
  });
}

export async function getQueueEntriesWithCount(queueId: string) {
  const [entries, count] = await Promise.all([
    getQueueEntries(queueId),
    db.queueEntry.count({
      where: { queueId, status: 'WAITING' },
    }),
  ]);

  return { entries, count };
}

export async function callNextEntry(queueId: string) {
  return db.$transaction(async (tx) => {
    // Get first waiting entry
    const nextEntry = await tx.queueEntry.findFirst({
      where: { queueId, status: 'WAITING' },
      orderBy: [{ position: 'asc' }, { createdAt: 'asc' }],
    });

    if (!nextEntry) {
      throw new Error('No waiting entries');
    }

    // Mark as served
    await tx.queueEntry.update({
      where: { id: nextEntry.id },
      data: { status: 'SERVED', servedAt: new Date() },
    });

    // Decrement positions of remaining entries
    await tx.queueEntry.updateMany({
      where: {
        queueId,
        position: { gt: nextEntry.position },
        status: 'WAITING',
      },
      data: {
        position: { decrement: 1 },
      },
    });

    return nextEntry;
  });
}

export async function removeEntry(entryId: string) {
  return db.queueEntry.delete({
    where: { id: entryId },
  });
}
```

### 2. Create Admin Queue Dashboard Page

**File:** `src/app/admin/queue/[queueId]/page.tsx`

```typescript
import { notFound, redirect } from 'next/navigation';
import { auth } from '@/lib/auth';
import { db } from '@/lib/prisma';
import { QueueManagement } from './queue-management';
import { QueueShareDialog } from '@/components/queue/queue-share-dialog';
import { formatQueueUrl } from '@/lib/utils';
import { Button } from '@/components/ui/button';
import { ArrowLeft } from 'lucide-react';

export default async function AdminQueuePage({
  params,
}: {
  params: { queueId: string };
}) {
  const session = await auth();
  if (!session?.user) {
    redirect('/login');
  }

  const queue = await db.queue.findUnique({
    where: { id: params.queueId },
    include: {
      entries: {
        where: { status: 'WAITING' },
        orderBy: [{ position: 'asc' }, { createdAt: 'asc' }],
      },
      _count: {
        select: {
          entries: {
            where: { status: 'WAITING' },
          },
        },
      },
    },
  });

  if (!queue) {
    notFound();
  }

  // Verify ownership
  if (queue.adminId !== session.user.id && session.user.role !== 'SUPER_ADMIN') {
    redirect('/unauthorized');
  }

  const queueUrl = formatQueueUrl(params.queueId);

  return (
    <div className="container mx-auto py-8 px-4">
      <div className="flex items-center gap-4 mb-8">
        <Button variant="ghost" size="icon" asChild>
          <a href="/dashboard">
            <ArrowLeft className="h-4 w-4" />
          </a>
        </Button>
        <div className="flex-1">
          <h1 className="text-3xl font-bold">{queue.name}</h1>
          <p className="text-muted-foreground">
            {queue._count.entries} people waiting
          </p>
        </div>
        <QueueShareDialog queueId={queue.id} queueUrl={queueUrl} />
      </div>

      <QueueManagement queue={queue} />
    </div>
  );
}
```

### 3. Create Queue Management Component

**File:** `src/app/admin/queue/[queueId]/queue-management.tsx`

```typescript
'use client';

import { useState } from 'react';
import { useQuery, useMutation, useQueryClient } from '@tanstack/react-query';
import { Card, CardContent } from '@/components/ui/card';
import { Button } from '@/components/ui/button';
import { Badge } from '@/components/ui/badge';
import { callNextEntry, getQueueEntries } from '@/lib/queue-entries';
import { Users, CheckCircle, UserX } from 'lucide-react';

interface QueueManagementProps {
  queue: {
    id: string;
    name: string;
    entries: Array<{
      id: string;
      firstName: string;
      lastName: string;
      phone: string;
      position: number;
    }>;
  };
}

export function QueueManagement({ queue: initialQueue }: QueueManagementProps) {
  const queryClient = useQueryClient();
  const [isCalling, setIsCalling] = useState(false);

  // Fetch queue entries with polling
  const { data: entries } = useQuery({
    queryKey: ['queue-entries', queue.id],
    queryFn: () => getQueueEntries(queue.id),
    initialData: initialQueue.entries,
    refetchInterval: 5000, // Poll every 5 seconds
  });

  // Call next entry mutation
  const callNextMutation = useMutation({
    mutationFn: () => callNextEntry(queue.id),
    onMutate: () => setIsCalling(true),
    onSuccess: () => {
      queryClient.invalidateQueries({ queryKey: ['queue-entries'] });
    },
    onSettled: () => setIsCalling(false),
  });

  const handleCallNext = async () => {
    if (entries.length === 0) {
      alert('No one is in queue');
      return;
    }
    await callNextMutation.mutateAsync();
  };

  return (
    <div className="space-y-6">
      {/* Action Bar */}
      <div className="flex justify-between items-center">
        <div className="flex items-center gap-2">
          <Users className="h-5 w-5" />
          <span className="font-semibold">
            {entries.length} Waiting
          </span>
        </div>
        <Button
          onClick={handleCallNext}
          disabled={entries.length === 0 || isCalling}
          size="lg"
        >
          <CheckCircle className="h-4 w-4 mr-2" />
          {isCalling ? 'Calling...' : 'Call Next'}
        </Button>
      </div>

      {/* Queue List */}
      <div className="grid gap-4">
        {entries.map((entry, index) => (
          <Card key={entry.id}>
            <CardContent className="flex items-center justify-between p-4">
              <div className="flex items-center gap-4">
                <Badge variant="secondary" className="text-lg px-3 py-1">
                  #{entry.position}
                </Badge>
                <div>
                  <p className="font-semibold">
                    {entry.firstName} {entry.lastName}
                  </p>
                  <p className="text-sm text-muted-foreground">
                    {entry.phone}
                  </p>
                </div>
              </div>
              <Button variant="outline" size="sm">
                <UserX className="h-4 w-4" />
              </Button>
            </CardContent>
          </Card>
        ))}

        {entries.length === 0 && (
          <Card>
            <CardContent className="py-8 text-center text-muted-foreground">
              <Users className="h-12 w-12 mx-auto mb-4 opacity-50" />
              <p>No one is in queue yet</p>
              <p className="text-sm">Share the QR code to invite people</p>
            </CardContent>
          </Card>
        )}
      </div>
    </div>
  );
}
```

### 4. Create React Query Provider

**File:** `src/components/providers/query-provider.tsx`

```typescript
'use client';

import { QueryClient, QueryClientProvider } from '@tanstack/react-query';
import { useState } from 'react';

export function QueryProvider({ children }: { children: React.ReactNode }) {
  const [queryClient] = useState(
    () =>
      new QueryClient({
        defaultOptions: {
          queries: {
            staleTime: 60 * 1000, // 1 minute
            refetchOnWindowFocus: true,
          },
        },
      })
  );

  return (
    <QueryClientProvider client={queryClient}>
      {children}
    </QueryClientProvider>
  );
}
```
