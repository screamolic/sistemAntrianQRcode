## Implementation Steps

### 1. Create Queue Schema Extension

**File:** `src/lib/queue.ts`

```typescript
import { db } from './prisma';
import { generateQueueId } from '@/lib/utils';

export async function createQueue(adminId: string, name?: string) {
  const queueId = generateQueueId(); // cuid()
  const expiresAt = new Date(Date.now() + 24 * 60 * 60 * 1000); // 24 hours

  return db.queue.create({
    data: {
      id: queueId,
      adminId,
      name: name || `Queue-${queueId.slice(0, 6)}`,
      expiresAt,
    },
  });
}

export async function getQueue(queueId: string) {
  return db.queue.findUnique({
    where: { id: queueId },
    include: {
      admin: { select: { id: true, name: true } },
      _count: { select: { entries: true } },
    },
  });
}

export async function isQueueExpired(queueId: string): Promise<boolean> {
  const queue = await getQueue(queueId);
  if (!queue) return true;
  return queue.expiresAt < new Date();
}
```

### 2. Create Queue Generation API/Server Action

**File:** `src/app/actions/queue.ts`

```typescript
'use server';

import { auth } from '@/lib/auth';
import { createQueue } from '@/lib/queue';
import { redirect } from 'next/navigation';

export async function createQueueAction(formData: FormData) {
  const session = await auth();
  if (!session?.user?.id) {
    throw new Error('Unauthorized');
  }

  const name = formData.get('name') as string || undefined;
  const queue = await createQueue(session.user.id, name);
  
  redirect(`/queue/${queue.id}`);
}
```

### 3. Create "Create Queue" Button/Modal

**File:** `src/components/queue/create-queue-button.tsx`

```typescript
'use client';

import { useState } from 'react';
import { Button } from '@/components/ui/button';
import {
  Dialog,
  DialogContent,
  DialogHeader,
  DialogTitle,
  DialogTrigger,
} from '@/components/ui/dialog';
import { Input } from '@/components/ui/input';
import { createQueueAction } from '@/app/actions/queue';

export function CreateQueueButton() {
  const [open, setOpen] = useState(false);
  const [pending, setPending] = useState(false);

  const handleSubmit = async (formData: FormData) => {
    setPending(true);
    try {
      await createQueueAction(formData);
    } finally {
      setPending(false);
      setOpen(false);
    }
  };

  return (
    <Dialog open={open} onOpenChange={setOpen}>
      <DialogTrigger asChild>
        <Button>Create New Queue</Button>
      </DialogTrigger>
      <DialogContent>
        <DialogHeader>
          <DialogTitle>Create Queue</DialogTitle>
        </DialogHeader>
        <form action={handleSubmit}>
          <Input
            name="name"
            placeholder="Queue name (optional)"
            className="mt-4"
          />
          <Button type="submit" className="mt-4" disabled={pending}>
            {pending ? 'Creating...' : 'Create Queue'}
          </Button>
        </form>
      </DialogContent>
    </Dialog>
  );
}
```

### 4. Add Utility Function for Queue ID Generation

**File:** `src/lib/utils.ts`

```typescript
import { cuid } from '@paralleldrive/cuid2';

export function generateQueueId() {
  return cuid();
}

export function formatQueueUrl(queueId: string) {
  const domain = process.env.NEXT_PUBLIC_APP_URL || 'http://localhost:3000';
  return `${domain}/queue/${queueId}`;
}
```
