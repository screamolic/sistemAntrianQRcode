## Implementation Steps

### 1. Create Evolution-API Docker Configuration

**File:** `docker-compose.evolution.yml`

```yaml
version: '3.8'

services:
  evolution-api:
    image: atendai/evolution-api:latest
    container_name: evolution-api
    ports:
      - "8080:8080"
    environment:
      - DATABASE_URL=postgresql://evolution:evolution@evolution-db:5432/evolution
      - API_KEY=${EVOLUTION_API_KEY:-your-secret-api-key}
      - DISABLE_MANAGER=false
    volumes:
      - evolution_data:/evolution/storage
    networks:
      - evolution-network
    restart: unless-stopped

  evolution-db:
    image: postgres:15-alpine
    container_name: evolution-postgres
    environment:
      - POSTGRES_USER=evolution
      - POSTGRES_PASSWORD=evolution
      - POSTGRES_DB=evolution
    volumes:
      - evolution_db_data:/var/lib/postgresql/data
    networks:
      - evolution-network
    restart: unless-stopped

volumes:
  evolution_data:
  evolution_db_data:

networks:
  evolution-network:
    driver: bridge
```

### 3. Create Setup Script

**File:** `scripts/setup-evolution.sh`

```bash
#!/bin/bash

# Generate random API key if not set
if [ -z "$EVOLUTION_API_KEY" ]; then
  EVOLUTION_API_KEY=$(openssl rand -hex 32)
  echo "Generated new API key: $EVOLUTION_API_KEY"
fi

# Start Evolution-API stack
docker-compose -f docker-compose.evolution.yml up -d

echo "Evolution-API starting..."
echo "Wait 30 seconds for initialization..."
sleep 30

# Check health
curl -s http://localhost:8080/health

echo ""
echo "Evolution-API is running!"
echo "Access manager at: http://localhost:8080/manager"
echo "API Key: $EVOLUTION_API_KEY"
```

### 4. Create Connection Helper

**File:** `src/lib/evolution/connection.ts`

```typescript
import { EvolutionAPI } from './client';

const API_URL = process.env.EVOLUTION_API_URL || 'http://localhost:8080';
const API_KEY = process.env.EVOLUTION_API_KEY || '';
const INSTANCE_NAME = process.env.EVOLUTION_INSTANCE_NAME || 'queue-automation';

export const evolution = new EvolutionAPI(API_URL, API_KEY, INSTANCE_NAME);

/**
 * Check if Evolution-API is connected
 */
export async function isEvolutionConnected(): Promise<boolean> {
  try {
    const status = await evolution.getInstanceStatus();
    return status.status === 'open';
  } catch {
    return false;
  }
}

/**
 * Get connection QR code for pairing
 */
export async function getConnectionQR() {
  return evolution.connect();
}
```

### 5. Create Evolution API Client

**File:** `src/lib/evolution/client.ts`

```typescript
export class EvolutionAPI {
  private baseUrl: string;
  private apiKey: string;
  private instanceName: string;

  constructor(baseUrl: string, apiKey: string, instanceName: string) {
    this.baseUrl = baseUrl;
    this.apiKey = apiKey;
    this.instanceName = instanceName;
  }

  private async request<T>(endpoint: string, options?: RequestInit): Promise<T> {
    const response = await fetch(`${this.baseUrl}${endpoint}`, {
      ...options,
      headers: {
        'Content-Type': 'application/json',
        apikey: this.apiKey,
        ...options?.headers,
      },
    });

    if (!response.ok) {
      throw new Error(`Evolution API error: ${response.statusText}`);
    }

    return response.json();
  }

  async getInstanceStatus() {
    return this.request<{ status: string }>(
      `/instance/connectionState/${this.instanceName}`
    );
  }

  async connect() {
    return this.request<{ base64: string; code: string }>(
      '/instance/connect',
      {
        method: 'POST',
        body: JSON.stringify({ instanceName: this.instanceName }),
      }
    );
  }

  async sendMessage(phone: string, message: string) {
    return this.request(
      '/message/sendText',
      {
        method: 'POST',
        body: JSON.stringify({
          instanceName: this.instanceName,
          number: phone,
          textMessage: { text: message },
        }),
      }
    );
  }
}
```
